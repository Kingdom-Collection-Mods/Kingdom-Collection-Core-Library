kc_add_splendor_small_effect = {
	if = {
		limit = {
			has_dlc = "Mandate of Heaven"
		}
		add_splendor = 100
	}
	else = {
		add_adm_power = 25
		add_dip_power = 25
		add_mil_power = 25
	}
}
